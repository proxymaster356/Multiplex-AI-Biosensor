let current = 0;

function renderDots() {
  const d = document.getElementById('dots');
  d.innerHTML = '';
  steps.forEach((_, i) => {
    const b = document.createElement('button');
    b.className = 'dot' + (i === current ? ' active' : '');
    b.title = steps[i].phase;
    b.onclick = () => { current = i; render(); };
    d.appendChild(b);
  });
}

function renderCard() {
  const s = steps[current];
  let vizHtml = '';
  if (s.vizType === 'bars' && s.viz) {
    vizHtml = `<div class="viz-box"><div style="font-size:11px;color:var(--color-text-tertiary);margin-bottom:8px;">${s.vizCaption}</div>`;
    s.viz.forEach(v => {
      vizHtml += `<div class="viz-row"><span style="font-size:12px;min-width:140px;color:var(--color-text-secondary);">${v.label}</span><div class="viz-bar-wrap"><div class="viz-bar" style="width:${v.val}%;background:${v.color};"></div></div><span style="font-size:12px;min-width:36px;text-align:right;color:var(--color-text-secondary);">${v.val}%</span></div>`;
    });
    vizHtml += '</div>';
  } else if (s.vizType === 'flow') {
    vizHtml = `<div class="viz-box" style="text-align:center;padding:16px;">
      <div style="display:flex;justify-content:center;gap:10px;align-items:center;flex-wrap:wrap;">
        <span class="tag tag-gray">Patient sample</span>
        <span style="color:var(--color-text-tertiary);font-size:14px;">&#8594;</span>
        <span class="tag tag-blue">AMR channel</span>
        <span class="tag tag-purple">Biofilm channel</span>
        <span class="tag tag-coral">Oncology channel</span>
      </div>
      <div style="font-size:11px;color:var(--color-text-tertiary);margin-top:10px;">${s.flowLabel}</div>
    </div>`;
  } else if (s.vizType === 'binding') {
    vizHtml = `<div class="viz-box">`;
    s.bindingItems.forEach(b => {
      vizHtml += `<div class="viz-row"><span style="font-size:12px;min-width:130px;">${b.label}</span><span class="tag ${b.ch==='AMR'?'tag-blue':b.ch==='Biofilm'?'tag-purple':'tag-coral'}" style="font-size:11px;">${b.ch} probe</span><span style="font-size:12px;color:#3B6D11;margin-left:8px;">&#10003; bound</span></div>`;
    });
    vizHtml += '</div>';
  } else if (s.vizType === 'filter') {
    vizHtml = `<div class="viz-box" style="font-size:12px;text-align:center;padding:16px;">
      <div style="display:flex;justify-content:center;align-items:center;gap:8px;flex-wrap:wrap;">
        <span class="tag tag-coral">Noisy raw signal</span>
        <span style="color:var(--color-text-tertiary);">&#8594;</span>
        <span class="tag tag-amber">Kalman filter</span>
        <span style="color:var(--color-text-tertiary);">&#8594;</span>
        <span class="tag tag-teal">Clean signal</span>
        <span style="color:var(--color-text-tertiary);">&#8594;</span>
        <span class="tag tag-green">Feature vector</span>
      </div>
      <div style="color:var(--color-text-tertiary);font-size:11px;margin-top:10px;">${s.filterCaption}</div>
    </div>`;
  }

  let tagsHtml = '<div class="tag-row">' + s.tags.map(t => `<span class="tag ${t.cls}">${t.label}</span>`).join('') + '</div>';

  document.getElementById('card').innerHTML = `
    <p class="step-label">${s.phase}</p>
    <p class="step-title">${s.title}</p>
    ${vizHtml}
    <p class="step-body">${s.body}</p>
    ${tagsHtml}
    <div class="nav-row">
      <button class="nav-btn" onclick="go(-1)" ${current===0?'disabled':''}>&#8592; Previous</button>
      <span class="step-counter">${current+1} of ${steps.length}</span>
      <button class="nav-btn" onclick="go(1)" ${current===steps.length-1?'disabled':''}>Next &#8594;</button>
    </div>
  `;
}

function go(dir) {
  current = Math.max(0, Math.min(steps.length-1, current+dir));
  render();
}

function render() {
  renderDots();
  renderCard();
}

render();
